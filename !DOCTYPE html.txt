<button onclick="submitData()">Submit</button>

<script>
const SCRIPT_URL = https://script.google.com/macros/s/AKfycbxT20IU0awkV23-9hRJBoEcXXQDe8D1FLkXSPSn618M_jNQcDEU9BVlO8tOwoSpnGUJ/exec;

function submitData() {
  const data = {
    client: document.getElementById("client").value,
    vehicle: document.getElementById("vehicle").value,
    location: document.getElementById("location").value,
    notes: document.getElementById("notes").value,
    obd: document.getElementById("obd").value,
    score: document.getElementById("score").value,
    verdict: getVerdict(document.getElementById("score").value),
    photos: ""
  };

  fetch(SCRIPT_URL, {
    method: "POST",
    body: JSON.stringify(data)
  })
  .then(() => alert("Saved!"));
}

function getVerdict(score) {
  score = parseInt(score);
  if (score <= 3) return "BUY";
  if (score <= 8) return "CAUTION";
  return "AVOID";
}
